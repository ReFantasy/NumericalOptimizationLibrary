85Matrix QuasiNewton::CorrectHk(Matrix Hk, Vector sk, Vector yk)
86{
87 if (_options->quasi_newton_type == QuasiNewtonSearchType::SR1)
88 {
89 // SR1
90 Vector tmp = sk - Hk * yk;
91 return Hk + (tmp * tmp.transpose()) / (tmp.transpose() * yk);
92 }
93 else if (_options->quasi_newton_type == QuasiNewtonSearchType::DFP)
94 {
95 // DFP
96 return Hk + (sk * sk.transpose()) / (sk.transpose() * yk) -
97 (Hk * yk * yk.transpose() * Hk) / (yk.transpose() * Hk * yk);
98 }
99 else if (_options->quasi_newton_type == QuasiNewtonSearchType::BFGS)
100 {
101 // BFGS
102 FLOAT d = yk.transpose() * sk;
103 FLOAT a = 1.0 + (FLOAT)(yk.transpose() * Hk * yk) / d;
104 Matrix b = (sk * sk.transpose()) / d;
105 Matrix c = sk * yk.transpose() * Hk + Hk * yk * sk.transpose();
106
107 return Hk + a * b - c / d;
108 }
109
110 throw std::invalid_argument("invalid quasi-newton type");
111}
