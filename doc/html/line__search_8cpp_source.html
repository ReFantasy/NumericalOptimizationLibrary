115FLOAT LineSearch::GoldenSection(FLOAT secton_a, FLOAT secton_b, FLOAT epsilon /*= 10e-3*/)
116{
117 FLOAT a = secton_a;
118 FLOAT b = secton_b;
119 static const FLOAT r = (std::sqrt(5) - 1) / 2; // 0.618
120
121 while ((b - a) > epsilon)
122 {
123 FLOAT al = a + (1.0 - r) * (b - a);
124 FLOAT ar = a + r * (b - a);
125
126 if (Phi(al) < Phi(ar))
127 {
128 b = ar;
129 }
130 else
131 {
132 a = al;
133 }
134 }
135
136 return (a + b) / 2.0;
