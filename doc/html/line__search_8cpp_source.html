46FLOAT LinearSearch::Armijo(FLOAT alpha, const Options &options)
47{
48
49 FLOAT rho = options.parameter_line_search_armijo_rho;
50
51 // p58
52 FLOAT phi0 = phi(0.0);
53 FLOAT dphi0 = dphi_da(0.0);
54 FLOAT last_alpha = 0;
55 int k = 0;
56 while (true)
57 {
58 FLOAT phi_alpha = phi(alpha);
59
60 if (phi_alpha < (phi0 + rho * dphi0 * alpha))
61 {
62 break;
63 }
64 FLOAT tmp = alpha;
65 alpha = Interpolation(last_alpha, alpha, k);
66 last_alpha = tmp;
67 k++;
68 }
69 return alpha;
70}

72FLOAT LinearSearch::CubicInterpolationMinimum(FLOAT last_alpha, FLOAT alpha)
73{
74 FLOAT c = dphi_da(0.0);
75 Matrix m(2, 2);
76 m << last_alpha * last_alpha, -alpha * alpha, -last_alpha * last_alpha * last_alpha, alpha * alpha * alpha;
77
78 Vector v(2);
79 v << (phi(alpha) - phi(0) - dphi_da(0) * alpha), (phi(last_alpha) - phi(0) - dphi_da(0) * last_alpha);
80 Vector res = (m * v) / (last_alpha * last_alpha * alpha * alpha * (alpha - last_alpha));
81 FLOAT a = res(0);
82 FLOAT b = res(1);
83 last_alpha = alpha;
84 alpha = (-b + std::sqrt(b * b - 3 * a * c)) / 3.0 / a;
85 return alpha;
86}

101FLOAT LinearSearch::Goldstein(FLOAT alpha, const Options &options)
102{
103 FLOAT a = 0.0;
104 FLOAT b = std::numeric_limits<FLOAT>::max();
105 FLOAT phi0 = phi(0.0);
106 FLOAT dphi0 = dphi_da(0.0);
107 int k = 0;
108
109 while (true)
110 {
111 FLOAT phi_alpha = phi(alpha);
112 if (phi_alpha > (phi0 + options.parameter_line_search_goldstein_p * dphi0 * alpha))
113 {
114 b = alpha;
115 alpha = (a + b) / 2.0;
116 k++;
117 continue;
118 }
119 if (phi_alpha < (phi0 + (1 - options.parameter_line_search_goldstein_p) * dphi0 * alpha))
120 {
121 a = alpha;
122 if (b == std::numeric_limits<FLOAT>::max())
123 alpha = a * 2;
124 else
125 alpha = (a + b) / 2.0;
126 k++;
127 continue;
128 }
129 break;
130 }

132 return alpha;

180void LinearSearch::AdvanceAndRetreat(FLOAT alpha0, FLOAT h0, FLOAT t, FLOAT &secton_a, FLOAT &secton_b)
181{
182 int i = 0;
183 FLOAT alpha, alpha_i, alpha_i_1, hi;
184 alpha = alpha_i = alpha0;
185 hi = h0;
186
187 FLOAT _a, _b;
188 while (true)
189 {
190 alpha_i_1 = alpha_i + hi;
191 if (alpha_i_1 <= 0)
192 {
193 alpha_i_1 = 0;
194 }
195 else if (phi(alpha_i_1) < phi(alpha_i))
196 {
197 // step 3
198 hi = t * hi;
199 alpha = alpha_i;
200 alpha_i = alpha_i_1;
201 i++;
202 continue;
203 }
204
205 // step 4
206 if (i == 0)
207 {
208 hi = -hi;
209 alpha = alpha_i_1;
210 i++;
211 }
212 else
213 {
214 _a = std::min(alpha, alpha_i_1);
215 _b = std::max(alpha, alpha_i_1);
216 break;
217 }
218 }
219
220 secton_a = _a;
221 secton_b = _b;
222}

227FLOAT LinearSearch::GoldenSection(FLOAT secton_a, FLOAT secton_b, FLOAT epsilon)
228{
229 FLOAT a = secton_a;
230 FLOAT b = secton_b;
231 static const FLOAT r = (std::sqrt(5) - 1) / 2; // 0.618
232
233 // TODO Optimization
234 while ((b - a) > epsilon)
235 {
236 FLOAT alpha_left = a + (1.0 - r) * (b - a);
237 FLOAT alpha_right = a + r * (b - a);
238
239 if (phi(alpha_left) < phi(alpha_right))
240 b = alpha_right;
241 else
242 a = alpha_left;
243 }
244
245 return (a + b) / 2.0;

253FLOAT LinearSearch::StrongWolfe(FLOAT alpha, const Options &options)
254{
255 FLOAT c1 = options.parameter_line_search_wolfe_rho;
256 FLOAT c2 = options.parameter_line_search_wolfe_sigma;
257 auto choose = [](FLOAT a, FLOAT b) {
258 if (std::max(a, b) != std::numeric_limits<FLOAT>::max())
259 {
260 return (a + b) / 2.0;
261 }
262 else
263 {
264 return std::min(a, b) * 2.0;
265 }
266 };
267
268 FLOAT a0 = 0;
269 FLOAT amax = options.parameter_line_search_wolfe_alpha_max;
270 FLOAT a1 = alpha;
271 if ((a1 < 0) || a1 >= amax)
272 a1 = choose(0, amax);
273
274 int i = 1;
275 while (true)
276 {
277 if ((phi(a1) > (phi(0) + c1 * a1 * dphi_da(0))) || ((phi(a1) >= phi(a0)) && (i > 1)))
278 return Zoom(a0, a1, options);
279 if (std::abs(dphi_da(a1)) <= (-c2 * dphi_da(0)))
280 return a1;
281 if (dphi_da(a1) >= 0)
282 return Zoom(a1, a0, options);
283
284 a0 = a1;
285 a1 = choose(a1, amax);
286 i++;
287 }
288}

290FLOAT LinearSearch::Zoom(FLOAT alo, FLOAT ahi, const Options &options)
291{
292 FLOAT c1 = options.parameter_line_search_wolfe_rho;
293 FLOAT c2 = options.parameter_line_search_wolfe_sigma;
294
295 while (true)
296 {
297 if (std::abs(ahi - alo) < options.min_step_size)
298 {
299 // assert(phi(ahi) < phi(0));
300 return std::max(ahi, alo);
301 }
302 FLOAT aj = (alo + ahi) / 2.0;
303 FLOAT tmp = phi(0) + c1 * aj * dphi_da(0);
304 if ((phi(aj) > tmp) || (phi(aj) >= phi(alo)))
305 {
306 ahi = aj;
307 }
308 else
309 {
310 FLOAT dphi_aj = dphi_da(aj);
311 if (std::abs(dphi_aj) <= (-c2 * dphi_da(0)))
312 return aj;
313 if ((dphi_aj * (ahi - alo)) >= 0)
314 ahi = alo;
315 alo = aj;
316 }
317 }
